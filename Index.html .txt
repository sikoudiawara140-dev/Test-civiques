function corriger() {
let score = 0;

questions.forEach((q, index) => {
let selected = document.querySelector(`input[name="q${index}"]:checked`);
if (selected) {
if (parseInt(selected.value) === q.answer) {
score++;
}
}
});

document.getElementById("resultat").innerText =
